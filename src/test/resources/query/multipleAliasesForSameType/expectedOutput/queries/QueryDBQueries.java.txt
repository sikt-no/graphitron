package fake.code.generated.queries.query;

import static no.sikt.graphitron.jooq.generated.testdata.FieldHelpers.*;
import static no.sikt.graphitron.jooq.generated.testdata.Keys.*;
import static no.sikt.graphitron.jooq.generated.testdata.Tables.*;
import static org.jooq.Functions.*;
import static org.jooq.impl.DSL.*;

import fake.graphql.example.package.model.Film;
import fake.graphql.example.package.model.Language;
import java.lang.String;
import java.util.List;
import no.fellesstudentsystem.graphql.helpers.selection.SelectionSet;
import org.jooq.DSLContext;

public class QueryDBQueries {
    public List<Film> filmForQuery(DSLContext ctx, String id, SelectionSet select) {
        var film_filmlanguage = LANGUAGE.as("film_filmlanguage");
        var film_originallanguage = LANGUAGE.as("film_originallanguage");
        return ctx
                .select(
                        row(
                                FILM.getId().as("id"),
                                row(
                                        film_filmlanguage.getId().as("id")
                                ).mapping(nullOnAllNull(Language::new)).as("filmLanguage"),
                                row(
                                        film_originallanguage.getId().as("id")
                                ).mapping(nullOnAllNull(Language::new)).as("originalLanguage")
                        ).mapping(nullOnAllNull(Film::new)).as("film")
                )
                .from(FILM)
                .leftJoin(film_filmlanguage)
                .onKey(FILM__FILM_LANGUAGE_ID_FKEY)
                .leftJoin(film_originallanguage)
                .onKey(FILM__FILM_ORIGINAL_LANGUAGE_ID_FKEY)
                .where(FILM.ID.eq(id))
                .orderBy(FILM.getIdFields())
                .fetch(0, Film.class);
    }
}
