package fake.code.example.package.queries.query;

import static no.fellesstudentsystem.kjerneapi.FieldHelpers.*;
import static no.fellesstudentsystem.kjerneapi.Keys.*;
import static no.fellesstudentsystem.kjerneapi.Tables.*;
import static org.jooq.Functions.*;
import static org.jooq.impl.DSL.*;

import java.lang.String;
import java.util.List;
import java.util.Map;
import java.util.Set;
import fake.graphql.example.package.model.Permisjon;
import fake.graphql.example.package.model.Permisjonsperiode;
import no.fellesstudentsystem.graphql.helpers.selection.SelectionSet;
import org.jooq.DSLContext;
import org.jooq.Record2;

public class StudentVedInstitusjonDBQueries {
    public Map<String, List<Permisjon>> permisjonerForStudentVedInstitusjon(
            DSLContext ctx, Set<String> studentVedInstitusjonIder, SelectionSet select) {
        return ctx
                .select(
                        PERMISJON.getStudentId(),
                        row(
                                PERMISJON.getId().as("id"),
                                row(
                                        select.optional("permisjonsperiode/fra", PERMISJON.DATO_FRA).as("fra"),
                                        select.optional("permisjonsperiode/til", PERMISJON.DATO_TIL).as("til")
                                ).mapping(nullOnAllNull(Permisjonsperiode::new)).as("permisjonsperiode")
                        ).mapping(nullOnAllNull(Permisjon::new)).as("permisjoner")
                )
                .from(PERMISJON)
                .where(PERMISJON.hasStudentIds(studentVedInstitusjonIder))
                .fetchGroups(Record2::value1, Record2::value2);
    }
}
