import fake.graphql.example.package.model.Language;

import org.jooq.Record2;

public class PaymentDBQueries {
    public Map<String, Language> originalLanguageForPayment(DSLContext ctx, Set<String> paymentIder,
            SelectionSet select) {
        var film_originallanguage = LANGUAGE.as("film_originallanguage");
        var payment_rental = PAYMENT.rental().as("payment_rental");
        var payment_rental_inventory = payment_rental.inventory().as("payment_rental_inventory");
        var payment_rental_inventory_film = payment_rental_inventory.film().as("payment_rental_inventory_film");
        return ctx
                .select(
                        PAYMENT.getId(),
                        row(
                                film_originallanguage.getId().as("id"),
                                select.optional("name", film_originallanguage.NAME).as("name")
                        ).mapping(nullOnAllNull(Language::new)).as("originalLanguage")
                )
                .from(PAYMENT)
                .join(film_originallanguage)
                .onKey(FILM__FILM_ORIGINAL_LANGUAGE_ID_FKEY)
                .where(PAYMENT.hasIds(paymentIder))
                .fetchMap(Record2::value1, Record2::value2);
    }
}
