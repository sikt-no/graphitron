package fake.code.generated.queries.query;

import static no.sikt.graphitron.jooq.generated.testdata.Tables.*;

import fake.graphql.example.package.model.Address;
import fake.graphql.example.package.model.Customer;
import java.lang.String;
import java.util.Map;
import java.util.Set;
import no.fellesstudentsystem.graphql.helpers.selection.SelectionSet;
import org.jooq.DSLContext;
import org.jooq.Record2;

public class CustomerDBQueries {
    public Map<String, Customer> loadCustomerByIdsAsNodeRef(DSLContext ctx, Set<String> ids,
            SelectionSet select) {
        return ctx
                .select(
                        CUSTOMER.getId(),
                        row(
                                CUSTOMER.getId().as("id"),
                                row(
                                        CUSTOMER.address().getId().as("id")
                                ).mapping(nullOnAllNull(Address::new)).as("address")
                        ).mapping(nullOnAllNull(Customer::new)).as("id")
                )
                .from(CUSTOMER)
                .where(CUSTOMER.hasIds(ids))
                .fetchMap(Record2::value1, Record2::value2);
    }
}
