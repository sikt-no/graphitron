import fake.graphql.example.package.model.Address;
import java.lang.String;
import java.util.Map;
import java.util.Set;
import no.fellesstudentsystem.graphql.helpers.selection.SelectionSet;
import org.jooq.DSLContext;
import org.jooq.Record2;

public class AddressDBQueries {
    public Map<String, Address> loadAddressByIdsAsNodeRef(DSLContext ctx, Set<String> ids,
            SelectionSet select) {
        return ctx
                .select(
                        ADDRESS.getId(),
                        row(
                                ADDRESS.getId().as("id")
                        ).mapping(nullOnAllNull(Address::new)).as("id")
                )
                .from(ADDRESS)
                .where(ADDRESS.hasIds(ids))
                .fetchMap(Record2::value1, Record2::value2);
    }
}
