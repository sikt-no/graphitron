import static no.fellesstudentsystem.kjerneapi.Tables.*;
import static org.jooq.Functions.*;

import java.util.Arrays;

import no.fellesstudentsystem.kjerneapi.tables.records.PersonRecord;

public class RegistrerPersonInputAndResponseDBQueries {
    public int registrerPersonInputAndResponse(DSLContext ctx, List<PersonRecord> inputRecordList) {
        return Arrays.stream(ctx.batchInsert(inputRecordList).execute()).sum();
    }
}
