package fake.code.example.package.resolvers.mutation;

import graphql.schema.DataFetchingEnvironment;
import java.lang.Exception;
import java.lang.Override;
import java.lang.String;
import java.util.concurrent.CompletableFuture;
import java.util.Map;
import javax.inject.Inject;
import fake.graphql.example.package.api.RegistrerPersonInputMutationResolver;
import fake.graphql.example.package.model.EndreInput;
import fake.graphql.example.package.model.Kjonn;
import fake.code.example.package.queries.mutation.RegistrerPersonInputDBQueries;
import no.fellesstudentsystem.graphql.helpers.arguments.Arguments;
import no.fellesstudentsystem.kjerneapi.FieldHelpers;
import no.fellesstudentsystem.kjerneapi.tables.records.PersonRecord;
import no.fellesstudentsystem.graphitron.enums.KjonnTest;
import org.jooq.DSLContext;

public class RegistrerPersonInputGeneratedResolver implements RegistrerPersonInputMutationResolver {
    @Inject
    DSLContext ctx;

    @Inject
    private RegistrerPersonInputDBQueries registrerPersonInputDBQueries;

    @Override
    public CompletableFuture<String> registrerPersonInput(EndreInput input, DataFetchingEnvironment env)
            throws Exception {
        var flatArguments = Arguments.flattenArgumentKeys(env.getArguments());

        var inputRecord = new PersonRecord();
        inputRecord.attach(ctx.configuration());

        if (input != null) {
            if (flatArguments.contains("input/privatEpost")) {
                inputRecord.setEmailadressePrivat(input.getPrivatEpost());
            }
            if (flatArguments.contains("input/id")) {
                inputRecord.setId(input.getId());
            }
            if (flatArguments.contains("input/fornavn")) {
                inputRecord.setFornavn(input.getFornavn());
            }
            if (flatArguments.contains("input/etternavn")) {
                inputRecord.setEtternavn(input.getEtternavn());
            }
            if (flatArguments.contains("input/kjonn")) {
                inputRecord.setKjonn(input.getKjonn() == null ? null : Map.of(Kjonn.MANN, KjonnTest.MANN, Kjonn.KVINNE, KjonnTest.KVINNE).getOrDefault(input.getKjonn(), null));
            }
            if (flatArguments.contains("input/statusEksporter")) {
                inputRecord.setStatusEksporterFlr(input.getStatusEksporter());
            }
            FieldHelpers.setPersonKeysFromPlattformIds(ctx, inputRecord);
        }

        var rowsUpdated = registrerPersonInputDBQueries.registrerPersonInput(inputRecord);

        return CompletableFuture.completedFuture(inputRecord.getId());
    }
}
