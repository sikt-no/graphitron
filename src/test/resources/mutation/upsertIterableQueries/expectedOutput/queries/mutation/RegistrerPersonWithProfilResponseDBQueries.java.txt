package fake.code.example.package.queries.mutation;

import static no.fellesstudentsystem.kjerneapi.FieldHelpers.*;
import static no.fellesstudentsystem.kjerneapi.Keys.*;
import static no.fellesstudentsystem.kjerneapi.Tables.*;
import static org.jooq.Functions.*;
import static org.jooq.impl.DSL.*;

import java.util.Arrays;
import java.util.List;
import javax.inject.Inject;
import no.fellesstudentsystem.kjerneapi.tables.records.PersonRecord;
import org.jooq.DSLContext;

public class RegistrerPersonWithProfilResponseDBQueries {
    @Inject
    DSLContext ctx;

    public int registrerPersonWithProfilResponse(List<PersonRecord> inputRecordList) {
        return Arrays.stream(ctx.batchStore(inputRecordList).execute()).sum();
    }
}
