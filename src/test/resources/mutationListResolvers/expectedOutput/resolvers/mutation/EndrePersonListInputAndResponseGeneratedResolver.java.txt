package no.fellesstudentsystem.graphql.resolvers.mutation;

import graphql.schema.DataFetchingEnvironment;
import java.lang.Exception;
import java.lang.Override;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.CompletableFuture;
import javax.inject.Inject;
import no.fellesstudentsystem.graphql.generated.api.EndrePersonListInputAndResponseMutationResolver;
import no.fellesstudentsystem.graphql.generated.model.EndreInput;
import no.fellesstudentsystem.graphql.generated.model.EndreResponse;
import no.fellesstudentsystem.graphql.helpers.selection.SelectionSets;
import no.fellesstudentsystem.kjerneapi.services.test_services.TestPersonService;
import no.fellesstudentsystem.kjerneapi.tables.records.PersonRecord;
import org.jooq.DSLContext;

public class EndrePersonListInputAndResponseGeneratedResolver implements EndrePersonListInputAndResponseMutationResolver {
    @Inject
    DSLContext ctx;

    @Override
    public CompletableFuture<List<EndreResponse>> endrePersonListInputAndResponse(
            List<EndreInput> input, DataFetchingEnvironment env) throws Exception {
        var testPersonService = new TestPersonService(ctx);
        var select = new SelectionSets(env.getSelectionSet());
        var inputRecordList = new ArrayList<PersonRecord>();


        if (input != null) {
            for (var itInput : input) {
                if (itInput == null) continue;
                var inputRecord = new PersonRecord();
                inputRecord.setId(itInput.getId());
                inputRecord.setFornavn(itInput.getFornavn());
                inputRecord.setEmailadressePrivat(itInput.getPrivatEpost());
                inputRecordList.add(inputRecord);
            }
        }

        var endrePersonListInputAndResponseResult = testPersonService.endrePersonListInputAndResponse(inputRecordList);

        var endreResponseList = new ArrayList<EndreResponse>();
        for (var itEndrePersonListInputAndResponseResult : endrePersonListInputAndResponseResult) {
            var endreResponse = new EndreResponse();
            endreResponse.setId(itEndrePersonListInputAndResponseResult.getId());
            endreResponse.setFornavn(itEndrePersonListInputAndResponseResult.getFornavn());
            endreResponse.setPrivatEpost(itEndrePersonListInputAndResponseResult.getEpostField());
            endreResponseList.add(endreResponse);
        }

        return CompletableFuture.completedFuture(endreResponseList);
    }
}
