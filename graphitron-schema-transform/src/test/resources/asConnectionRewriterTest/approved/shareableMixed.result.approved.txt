type OtherType {
  id: ID!
}

type PageInfo @shareable {
  hasPreviousPage: Boolean!
  hasNextPage: Boolean!
  startCursor: String
  endCursor: String
}

type Query {
  someType(first: Int = 100, after: String): QuerySomeTypeConnection @shareable
  otherType(first: Int = 100, after: String): QueryOtherTypeConnection
}

type QueryOtherTypeConnection {
  edges: [QueryOtherTypeConnectionEdge]
  pageInfo: PageInfo
  nodes: [OtherType]
  totalCount: Int
}

type QueryOtherTypeConnectionEdge {
  cursor: String
  node: OtherType
}

type QuerySomeTypeConnection @shareable {
  edges: [QuerySomeTypeConnectionEdge]
  pageInfo: PageInfo
  nodes: [SomeType]
  totalCount: Int
}

type QuerySomeTypeConnectionEdge @shareable {
  cursor: String
  node: SomeType
}

type SomeType {
  id: ID!
}
